31/// an enumeration of errors that this crate can produce
320#[derive(thiserror::Error, Debug)]
33pub enum Error {
34 /// An error from serializing or deserializing from a `Document`.
35 #[error("error working with storage: {0}")]
36 Storage(String),
37
38 /// An attempt to use a `Collection` with a `Database` that it wasn't defined within.
39 #[error("attempted to access a collection not registered with this schema")]
40 CollectionNotFound,
41
42 // TODO consider moving these to something like a document::Error
43 /// An attempt to update a document that doesn't exist.
44 #[error("the requested document id {1} from collection {0} was not found")]
45 DocumentNotFound(collection::Id, u64),
46
47 /// When updating a document, if a situation is detected where the contents have changed on the server since the `Revision` provided, a Conflict error will be returned.
48 #[error("a conflict was detected while updating document id {1} from collection {0}")]
49 DocumentConflict(collection::Id, u64),
50}
